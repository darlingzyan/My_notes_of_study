#### lowbit定义

lowbit(n)定义为非负整数n在二进制表示下“最低位的1及其后边所有的0”

```cpp
lowbit(n) = n & (~n+1) = n & (-n);
```

树状数组是一种基于上述思想的数据结构，其基本用途是**维护序列的前缀和**。对于给定的序列a，我们建立一个数组c，其中c[x]**保存序列a的区间[x-lowbit(x)+1,x]中所有数的和**

模板如下<br>

***Code***

```cpp
int ask(int x){
	int ans = 0;
	for(;x;x-= x&(-x)) ans+=c[x];
	return ans;
}
```

```cpp
void add(int x,int y){
	for(;x<=n;x+= x&(-x)) c[x]+=y;
}
```

```cpp
#include <bits/stdc++.h>
#define int long long
using namespace std;
const int N = 3e5+10;

int test, n, m;
int f[N];

void clear() {
	memset(f, 0, sizeof(f));
}

void add(int x, int y) {
	for (; x <= n; x += x & (-x)) f[x] += y;
}

int ask(int x) {
	int res = 0;
	for (; x; x -= x & (-x)) res += f[x];
	return res;
}

void solve() {
	cin >> n >> m;
	if (n == 1) {
		cout << "0\n";
		return;
	}
	vector<pair<int, int>> a(n + 1);
	for (int i = 1; i <= n; i++) {
		cin >> a[i].first;
		a[i].second = i;
	}

	sort(a.begin() + 1, a.end(), [](const auto & aa, const auto & bb) {
		return aa.first < bb.first;
	});

	vector<pair<int, int>> pairs;

	//处理首情况
	int l = min(a[1].second, a[2].second), r = max(a[1].second, a[2].second);
	pairs.push_back({l, r});

	//处理尾情况
	l = min(a[n].second, a[n - 1].second), r = max(a[n].second, a[n - 1].second);
	pairs.push_back({l, r});

	for (int i = 2; i < n; i++) {
		int ldif = a[i].first - a[i - 1].first;
		int rdif = a[i + 1].first - a[i].first;
		if (ldif < rdif) {
			int x = a[i].second, y = a[i - 1].second;
			if (x > y) swap(x, y);
			pairs.push_back({x, y});
		} else if (ldif == rdif) {
			int x1 = a[i].second, y1 = a[i - 1].second;
			int x2 = a[i].second;
			int y2 = a[i + 1].second;
			if (x1 > y1) swap(x1, y1);
			if (x2 > y2) swap(x2, y2);
			pairs.push_back({x1, y1});
			pairs.push_back({x2, y2});
		} else {
			int x = a[i].second, y = a[i + 1].second;
			if (x > y) swap(x, y);
			pairs.push_back({x, y});
		}
	}

	sort(pairs.begin(), pairs.end(), [](const auto & aa, const auto & bb) {
		if (aa.second != bb.second) return aa.second < bb.second;
		return aa.first < bb.first;
	});

	int ans = 0;
	int pair_idx = 0;
	int total_pairs = pairs.size();

	vector<pair<int, pair<int, int> > > dicar;
	for (int i = 1; i <= m; i++) {
		int l, r;
		cin >> l >> r;
		dicar.push_back({i, {l, r}});
	}

	sort(dicar.begin(), dicar.end(), [](const auto & aa, const auto & bb) {
		return aa.second.second < bb.second.second;
	});

	for (auto &it : dicar) {
		while (pair_idx < total_pairs && pairs[pair_idx].second <= it.second.second) {
			add(pairs[pair_idx].first, 1);
			pair_idx++;
		}
		int lef = ask(it.second.first - 1);
		int cnt = pair_idx - lef;
		ans += cnt * it.first;
	}

	cout << ans << '\n';
}
```

### 归并排序

```cpp
#include <bits/stdc++.h>
#define int long long
using namespace std;

const int N = 200005;
int a[N], tmp[N]; // tmp 是归并排序过程中的辅助数组

// 归并排序求逆序对
// 区间范围：[l, r]
int merge_sort(int l, int r) {
    if (l >= r) return 0; // 递归边界：区间内只有一个元素或为空，逆序对为 0

    int mid = l + r >> 1;
    // 1. 递归求解左半段和右半段内部的逆序对
    int res = merge_sort(l, mid) + merge_sort(mid + 1, r);

    // 2. 归并过程：统计横跨左右两半段的逆序对
    int i = l, j = mid + 1, k = 0;
    while (i <= mid && j <= r) {
        if (a[i] <= a[j]) {
            tmp[k++] = a[i++]; // 左边较小，不构成逆序对
        } else {
            // 右边较小，构成逆序对！
            // 因为左半段 [i, mid] 是有序的，如果 a[i] > a[j]，
            // 那么 a[i] 到 a[mid] 的所有元素都大于 a[j]
            res += mid - i + 1; 
            tmp[k++] = a[j++];
        }
    }

    // 3. 把还没有比较完的剩余元素直接放入辅助数组
    while (i <= mid) tmp[k++] = a[i++];
    while (j <= r) tmp[k++] = a[j++];

    // 4. 将排好序的辅助数组拷贝回原数组
    for (i = l, k = 0; i <= r; i++, k++) {
        a[i] = tmp[k];
    }

    return res;
}

void solve() {
    int n;
    cin >> n;
    for (int i = 1; i <= n; i++) {
        cin >> a[i];
    }

    // 调用函数并输出答案
    cout << merge_sort(1, n) << "\n";
}

signed main() {
    ios::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);

    int test;
    if (cin >> test) {
        while (test--) {
            solve();
        }
    }
    return 0;
}
```

### 离散化

不必多说上代码模板

***Code***

```cpp
#include <bits/stdc++.h>
#define int long long
using namespace std;

void solve() {
    int n;
    cin >> n;
    vector<int> a(n);
    for (int i = 0; i < n; i++) cin >> a[i];

    // --- 离散化开始 ---
    vector<int> b = a; // 1. 复制一份

    sort(b.begin(), b.end()); // 2. 排序
    
    // 3. 去重 (unique会将不重复的元素移到前面，返回去重后最后一个元素的迭代器)
    b.erase(unique(b.begin(), b.end()), b.end()); 

    // 4. 映射回原数组 (注意下标加 1，因为树状数组下标必须从 1 开始)
    for (int i = 0; i < n; i++) {
        // lower_bound 返回指向找到的元素的迭代器，减去 b.begin() 就得到排名索引
        a[i] = lower_bound(b.begin(), b.end(), a[i]) - b.begin() + 1; 
    }
    // --- 离散化结束 ---

    // 现在的 a 数组已经变成了排名 (1 到 n 之间)，可以直接扔进树状数组求逆序对了
    /*
       for (int i = n - 1; i >= 0; i--) { ... }
    */
}
```

## 数学知识

### 异或

#### 连续整数异或和的周期性质

* **核心数学规律：连续异或和**

  

$首先，我们定义前缀异或和 P(i) = 1 \oplus 2 \oplus 3 \oplus .....\oplus i , 并规定P(0) = 0 ， 如果要求区间[l,r] 异或和为0 且包含X，等价于：$

1. **$P(l-1) \oplus P(r) = 0$**，即 $P(l-1) = P(r)$。

2. **$1 \le l \le x \le r \le n$**。<br>

根据连续整数异或的性质，$P(i)$ 的取值有 $4$ 为周期的规律：

* $P(i) = i$，若 $i \equiv 0 \pmod 4$

  $P(i) = 1$，若 $i \equiv 1 \pmod 4$

  $P(i) = i+1$，若 $i \equiv 2 \pmod 4$

  $P(i) = 0$，若 $i \equiv 3 \pmod 4$

  

然后分类讨论即可
